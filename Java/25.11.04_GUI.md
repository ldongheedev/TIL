package sec04;

import java.awt.*;
import javax.swing.*;
import java.awt.event.*;

//내부 클래스로 Action이벤트 리스트 처리



public class EventTest2 extends JFrame{
	
	JButton btn1 ;
	JButton btn2 ;
	JButton btnC;
	
	EventTest2() {
		setTitle("액션 이벤트 리스너 예제");
		setLayout(new FlowLayout());
		
		btn1 = new JButton("확인");
		btn2 = new JButton("취소");
		
		btnC = new JButton("색_변경");

		add(btn1);
		add(btn2);
		add(btnC);
		
		btn1.addActionListener(new MyGUI1());
		btn2.addActionListener(new MyGUI2());
		btnC.addActionListener(new MyGUI_C());
		
		
		
		setSize(300, 150);
		setVisible(true);
	}
	
	//내부 클래스 정의
	class MyGUI1 implements ActionListener {
		@Override
		public void actionPerformed(ActionEvent e) {
			setTitle("확인을 눌렀습니다.");
			String newText = btn1.getText().equals("확인") ? "OK" : "확인";
			btn1.setText(newText);
		}
	}

	class MyGUI2 implements ActionListener {
		@Override
		public void actionPerformed(ActionEvent e) {
			setTitle("취소가 눌렸습니다.");
			String newText = btn2.getText().equals("취소") ? "cancle" : "취소";
			btn2.setText(newText);
		}
	}
	
	class MyGUI_C implements ActionListener {

		@Override
		public void actionPerformed(ActionEvent e) {
			
			if(btn1.getBackground().equals(Color.yellow)) {
				btn1.setBackground(Color.white);
				btn2.setBackground(Color.white);
			}else {
				btn1.setBackground(Color.yellow);
				btn2.setBackground(Color.green);	
			}
			
		}
		
	}//class
	

	public static void main(String[] args) {
		new EventTest2();
	}

}
